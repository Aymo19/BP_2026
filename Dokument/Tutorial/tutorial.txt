Prerekvizity
	GCC a G++:
		sudo apt install gcc-11 g++-11
		sudo update-alternatives --install /usr/bin/gcc gcc /usr/bin/gcc-11 11
		sudo update-alternatives --install /usr/bin/g++ g++ /usr/bin/g++-11 11
		sudo update-alternatives --config gcc
		sudo update-alternatives --config g++
	
	VOLK:
		git clone --recursive https://github.com/gnuradio/volk.git
    	cd volk
    	mkdir build
    	cd build
		cmake -DCMAKE_BUILD_TYPE=Release -DPYTHON_EXECUTABLE=/usr/bin/python3 ../
    	make
    	make test
    	sudo make install
		sudo ldconfig

Postup istalacie GR
	git clone https://github.com/gnuradio/gnuradio.git
	cd gnuradio
    git checkout maint-X (kde X je verzia GR, napr 3.8)
    git submodule update --init --recursive
    mkdir build
    cd build
	
	cmake -DCMAKE_BUILD_TYPE=Release -DPYTHON_EXECUTABLE=/usr/bin/python3 ../
	make -jX (kde X je pocet CPU, napr 4)
	sudo make install
	sudo ldconfig

Spustenie
	Ak sa nevytvori odkaz na GR-Companion tak treba to spustit cez Terminal.
		gnuradio-companion

Debug
	ModuleNotFoundError
		gnuradio-config-info --prefix (default /user/local)
		find /user/local -name gnuradio | grep "packages"
*		export PYTHONPATH=/user/local/lib/python3/dist-packages:$PYTHONPATH
*		export LD_LIBRARY_PATH=/usr/local/lib:$LD_LIBRARY_PATH

Prikazy * je odporucane si hodit do startup scriptu
